// ─── Website Scanner ─────────────────────────────────────────────────────────
const CLIENT_PROFILES_FILE = resolve('data/client-profiles.json');
let clientProfiles = new Map();
try {
  const raw = JSON.parse(readFileSync(CLIENT_PROFILES_FILE, 'utf8'));
  for (const [k, v] of Object.entries(raw)) clientProfiles.set(k, v);
} catch {}
function persistProfiles() {
  writeFileSync(CLIENT_PROFILES_FILE, JSON.stringify(Object.fromEntries(clientProfiles), null, 2));
}

async function scanWebsite(url) {
  const pages = [];
  try {
    // Fetch homepage
    const homeRes = await fetch(url);
    const homeHtml = await homeRes.text();
    pages.push({ url, html: homeHtml });

    // Find internal links to key pages
    const linkPattern = /<a[^>]+href=["']([^"'#]+)["'][^>]*>([^<]*)<\/a>/gi;
    const keywords = ['about', 'services', 'contact', 'menu', 'pricing', 'team', 'faq'];
    const found = new Set();
    let match;
    while ((match = linkPattern.exec(homeHtml)) !== null && found.size < 5) {
      const href = match[1];
      const text = match[2].toLowerCase();
      if (keywords.some(k => href.toLowerCase().includes(k) || text.includes(k))) {
        let fullUrl = href;
        if (href.startsWith('/')) fullUrl = new URL(href, url).href;
        else if (!href.startsWith('http')) fullUrl = new URL(href, url).href;
        if (!found.has(fullUrl)) {
          found.add(fullUrl);
          try {
            const r = await fetch(fullUrl);
            pages.push({ url: fullUrl, html: await r.text() });
          } catch {}
        }
      }
    }
  } catch (e) {
    throw new Error(`Could not reach ${url}: ${e.message}`);
  }

  // Strip HTML to text
  const stripped = pages.map(p => {
    const text = p.html
      .replace(/<script[^>]*>[\s\S]*?<\/script>/gi, '')
      .replace(/<style[^>]*>[\s\S]*?<\/style>/gi, '')
      .replace(/<[^>]+>/g, ' ')
      .replace(/\s+/g, ' ')
      .trim()
      .slice(0, 3000);
    return `PAGE: ${p.url}\n${text}`;
  }).join('\n\n---\n\n');

  // Ask Claude to extract business profile
  const msg = await claude.messages.create({
    model: 'claude-haiku-4-5-20251001',
    max_tokens: 1000,
    messages: [{ role: 'user', content: `Analyze this website content and extract a business profile. Return ONLY valid JSON with these fields:
{
  "name": "business name",
  "services": "comma-separated list of services",
  "location": "city/area",
  "phone": "phone number or empty string",
  "email": "email or empty string",
  "hours": "working hours or empty string",
  "tone": "professional" or "friendly" or "casual",
  "summary": "one sentence describing what this business does"
}

Website content:
${stripped}` }],
  });

  try {
    const jsonStr = msg.content[0].text.replace(/```json\n?|```/g, '').trim();
    return JSON.parse(jsonStr);
  } catch {
    return { name: '', services: '', location: '', phone: '', email: '', hours: '', tone: 'friendly', summary: '' };
  }
}
```

app.post('/api/scan-website', async (req, res) => {
  const { url } = req.body;
  if (!url) return res.status(400).json({ error: 'URL required' });

  // Cache check
  const cached = scanCache.get(url);
  if (cached && Date.now() - cached.time < 86400000) return res.json(cached.profile);

  try {
    const profile = await scanWebsite(url);
    scanCache.set(url, { profile, time: Date.now() });
    res.json(profile);
  } catch (e) {
    res.status(500).json({ error: e.message });

});

// ─── Invite System ───────────────────────────────────────────────────────────
app.post('/api/admin/invite', (req, res) => {
  if (req.query.pass !== ADMIN && req.headers['x-admin-pass'] !== ADMIN) return res.status(401).json({ error: 'Unauthorized' });
  const { email, url, type } = req.body;
  if (!email) return res.status(400).json({ error: 'Email required' });

  const token = crypto.randomBytes(16).toString('hex');
  invites.set(token, {
    email: email.toLowerCase(),
    url: url || '',
    type: type || '',
    createdAt: new Date().toISOString(),
    used: false,
  });
  persistInvites();

  const serverUrl = process.env.RAILWAY_PUBLIC_DOMAIN
    ? `https://${process.env.RAILWAY_PUBLIC_DOMAIN}`
    : `http://localhost:${process.env.PORT || 3000}`;
  const link = `${serverUrl}/onboard?t=${token}`;
  res.json({ ok: true, token, link });
});

app.get('/api/admin/invites', (req, res) => {
  if (req.query.pass !== ADMIN) return res.status(401).json({ error: 'Unauthorized' });
  const list = [];
  for (const [token, inv] of invites) {
    // Expire after 7 days
    if (!inv.used && Date.now() - new Date(inv.createdAt).getTime() > 7 * 86400000) {
      inv.expired = true;
    }
    list.push({ token, ...inv });
  }
  res.json(list);
});

app.delete('/api/admin/invite/:token', (req, res) => {
  if (req.query.pass !== ADMIN) return res.status(401).json({ error: 'Unauthorized' });
  invites.delete(req.params.token);
  persistInvites();
  res.json({ ok: true });
});
```

// ─── File Upload Chat ────────────────────────────────────────────────────────
app.post('/api/chat/upload', async (req, res) => {
  try {
    const { image, message, system, sessionId, model } = req.body;
    // image is base64 data URL: "data:image/jpeg;base64,..."
    if (!image) return res.status(400).json({ error: 'No image provided' });

    const match = image.match(/^data:(image\/\w+);base64,(.+)$/);
    if (!match) return res.status(400).json({ error: 'Invalid image format' });

    const mediaType = match[1];
    const data = match[2];

    const messages = [{
      role: 'user',
      content: [
        { type: 'image', source: { type: 'base64', media_type: mediaType, data } },
        { type: 'text', text: message || 'What do you see in this image? How can you help based on what you see?' },
      ],
    }];

    const response = await claude.messages.create({
      model: model || 'claude-haiku-4-5-20251001',
      max_tokens: 800,
      system: system || '',
      messages,
    });

    trackUsage(response.usage?.input_tokens || 0, response.usage?.output_tokens || 0);
    res.json(response);
  } catch (e) {
    res.status(500).json({ error: e.message });
  }
});

function speakText(text, btn) {
  if (!window.speechSynthesis) return;
  // Stop any current speech
  speechSynthesis.cancel();
  const utter = new SpeechSynthesisUtterance(text.replace(/[#*_`]/g, ''));
  utter.rate = 1;
  utter.pitch = 1;
  btn.textContent = '⏹';
  utter.onend = () => { btn.textContent = '🔊'; };
  speechSynthesis.speak(utter);
}

```javascript
async function generateInvite() {
  const email = document.getElementById('inv-email').value;
  if (!email) return alert('Email required');
  const r = await fetch('/api/admin/invite?pass=' + ADMIN, {
    method: 'POST', headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({
      email,
      url: document.getElementById('inv-url').value,
      type: document.getElementById('inv-type').value,
    }),
  });
  const d = await r.json();
  if (d.ok) {
    document.getElementById('inv-link').value = d.link;
    document.getElementById('inv-result').style.display = 'block';
    loadInvites();
  }
}

async function loadInvites() {
  const r = await fetch('/api/admin/invites?pass=' + ADMIN);
  const invites = await r.json();
  const el = document.getElementById('inv-list');
  if (!invites.length) { el.innerHTML = '<p style="color:#6b6b8a;">No invites yet</p>'; return; }
  el.innerHTML = '<table style="width:100%;font-size:13px;"><tr><th>Email</th><th>Status</th><th>Created</th><th></th></tr>' +
    invites.map(i => {
      const status = i.used ? '✅ Used' : i.expired ? '⏰ Expired' : '🟢 Active';
      const created = new Date(i.createdAt).toLocaleDateString('en-GB');
      return `<tr><td>${i.email}</td><td>${status}</td><td>${created}</td><td>${!i.used ? `<button onclick="deleteInvite('${i.token}')" style="background:none;border:none;color:#ff6b6b;cursor:pointer;">Delete</button>` : ''}</td></tr>`;
    }).join('') + '</table>';
}

async function deleteInvite(token) {
  await fetch('/api/admin/invite/' + token + '?pass=' + ADMIN, { method: 'DELETE' });
  loadInvites();
}
```
